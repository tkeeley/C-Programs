#include <cs50.h>
#include <stdio.h>
#include <string.h>
#include <ctype.h>
#include <stdlib.h>


int main(int argc, char* argv[])
{
    // get key from command line argument, return 1 if error
    if (argc < 2)
    {
        printf("No command line argument entered!\n");
        return 1;
    }
    
    // change the number from a string to an integer
    int k = atoi(argv[1]);
    
    // check for error
    if (k < 1)
    {
        printf("Bad inputs!\n");
        return 1;
    }
    
    // get users string
    char* text = GetString();
    
    // print cipher
    for (int i = 0, n = strlen(text); i < n; i++)
    {
        if (isalpha(text[i]))
        {
            // for upper case letters
            if (isupper(text[i]))
            {
                int result = (text[i] + k - 65) % 26 + 65;
                printf("%c", result);
            }
            // for lower case letters
            else
            {
                int result = (text[i] + k - 97) % 26 + 97;
                printf("%c", result);
            }
        }
        // for other chars
        else
            printf("%c", text[i]);
    }
    // new line
    printf("\n");
}
